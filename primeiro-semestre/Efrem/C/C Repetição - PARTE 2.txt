#include <stdio.h>

int main() {
    int i, n, divisores;

    printf("Digite um número inteiro e natural: ");
    scanf("%d", &n);

    if (n <= 1) {
        printf("Erro: número deve ser maior que 1");
    } else {
        divisores = 0;

        for (i = 1; i <= n; i++) {
            if (n % i == 0) {
                divisores = divisores + 1;
            }
        }

        if (divisores == 2) {
            printf("O número %d é primo\n", n);
        } else {
            printf("O número %d não é primo\n", n);
        }
    }

    return 0;
}

#include <stdio.h>

int main() {
    int i, valor, vista, prazo, total;
    char codigo;

    vista = 0;
    prazo = 0;

    for (i = 1; i <= 15; i++) {
        printf("Digite o código da compra: ");
        scanf(" %c", &codigo);
        printf("Digite o valor da compra: ");
        scanf("%d", &valor);

        if (codigo == 'V' || codigo == 'v') {
            vista = vista + valor;
        } else if (codigo == 'P' || codigo == 'p') {
            prazo = prazo + valor;
        } else {
            printf("Código inválido! Transação ignorada.\n");
        }
    }

    total = vista + prazo;

    printf("Total das compras à vista: R$ %d\n", vista);
    printf("Total das compras a prazo: R$ %d\n", prazo);
    printf("Total geral das compras: R$ %d\n", total);

    return 0;
}

#include <stdio.h>

int main() {
    int i, filhos, porcent, mediaFilho;
    float sal, maiorSal, mediaSal;

    maiorSal = 0;
    mediaSal = 0;
    mediaFilho = 0;
    porcent = 0;

    for (i = 1; i <= 200; i++) {
        printf("Quanto é o seu salário?: ");
        scanf("%f", &sal);
        printf("Quantos filhos você tem?: ");
        scanf("%d", &filhos);

        mediaSal = mediaSal + sal;
        mediaFilho = mediaFilho + filhos;

        if (i == 1) {
            maiorSal = sal;
        }

        if (sal > maiorSal) {
            maiorSal = sal;
        }

        if (sal <= 150) {
            porcent = porcent + 1;
        }
    }

    printf("Média dos salários: R$ %.2f\n", mediaSal / 200);
    printf("Média de filhos: %.2f\n", (float)mediaFilho / 200);
    printf("Maior salário: R$ %.2f\n", maiorSal);
    printf("Porcentagem de pessoas com salário até R$ 150,00: %.2f%%\n", (porcent / 200.0) * 100);

    return 0;
}

#include <stdio.h>

int main() {
    int i, carteira, numMultas, maiorNumMultas = 0, maisMultado = 0, continuar;
    float multa, divida, total = 0;

    do {
        divida = 0;
        printf("Digite o número da carteira de motorista (1 a 4327): ");
        scanf("%d", &carteira);
        printf("Digite o número de multas: ");
        scanf("%d", &numMultas);

        for (i = 1; i <= numMultas; i++) {
            printf("Digite o valor da multa %d: R$ ", i);
            scanf("%f", &multa);
            divida = divida + multa;
        }

        total = total + divida;

        if (numMultas > maiorNumMultas) {
            maiorNumMultas = numMultas;
            maisMultado = carteira;
        }

        printf("Deseja continuar (1 = sim | 0 = não): ");
        scanf("%d", &continuar);

    } while (continuar == 1);

    printf("\nMotorista da carteira %d deve R$ %.2f\n", carteira, divida);
    printf("Total arrecadado com multas: R$ %.2f\n", total);
    printf("Carteira com maior número de multas: %d (%d multas)\n", maisMultado, maiorNumMultas);

    return 0;
}
