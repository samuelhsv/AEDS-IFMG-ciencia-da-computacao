int main() {
    int n1, n2, n3, menor, meio, maior;
    
    printf("Digite o número 1: ");
    scanf("%d", &n1);
    printf("Digite o número 2: ");
    scanf("%d", &n2);
    printf("Digite o número 3: ");
    scanf("%d", &n3);
    
    if ((n1 > n2) && (n1 > n3)){ // n1 é o maior
        maior = n1;
        if (n2 > n3){
          meio = n2;
          menor = n3;
        } else {
            meio = n3;
            menor = n2;
        }
    } else if ((n1 > n2) || (n1 > n3)){ // n1 é o do meio
        meio = n1;
        if (n2 > n3) {
            maior = n2;
            menor = n3;
        } else {
            maior = n3;
            menor = n2;
        }
    } else { // n1 é o menor
        menor = n1;
        if (n2 > n3){
            maior = n2;
            meio = n3;
        } else {
            maior = n3;
            meio = n2;
        }
    }
    
    printf("Maior: %d\n", maior);
    printf("Meio: %d\n", meio);
    printf("Menor: %d\n", menor);
    
    return 0;
}

int main() {
    char oper;
    int n1, n2;
    
    printf("Digite o 1° número: ");
    scanf("%d", &n1);
    printf("Digite o 2° número: ");
    scanf("%d", &n2);
    printf("Selecione a operação (+|-|*|/): ");
    scanf(" %c", &oper);
    
    
    if (oper == '+'){
        printf("A soma dos números %d e %d é %d", n1, n2, n1 + n2);    
    } else if (oper == '-') {
        printf("A subtração dos números %d e %d é %d", n1, n2, n1 - n2);
    } else if (oper == '*') {
        printf("O produto dos números %d e %d é %d", n1, n2, n1 * n2);
    } else if (oper == '/') {
        printf("A divisão dos números %d e %d é %d", n1, n2, n1 / n2);
    } else{
        printf("Erro: operaçãõ inválida");
    }
    
    return 0;
}
