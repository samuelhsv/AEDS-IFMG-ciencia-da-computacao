#include <stdio.h>

int main() {
    int i, n, pos, neg, zero;
   
   pos = 0;
   neg = 0;
   zero = 0;
   
    for (i=1; i<=10; i++){
        printf("Digite um numero: ");
        scanf("%d", &n);
        
        if (n > 0){
            pos++;
        } else if (n < 0) {
            neg++;
        } else {
            zero++;
        }
    }
       
    printf("Positivos: %d\nNegativos: %d\nZeros: %d", pos, neg, zero);
       
    return 0;
}

            if (primeiro){
