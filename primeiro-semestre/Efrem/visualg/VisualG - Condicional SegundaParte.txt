6.
Algoritmo "OrdemCrescente"

var
   a, b, c, menor, meio, maior: inteiro

inicio
   escreva("Digite o 1° número: ")
   leia(a)
   escreva("Digite o 2° número: ")
   leia(b)
   escreva("Digite o 3° número: ")
   leia(c)

   se (a > b) e (a > c) entao // 1° caso: se A for maior que B e C
      maior <- a
      se (b > c) entao // (1° caso) Disputa de meio e menor
         meio <- b
         menor <- c
      senao
         meio <- c
         menor <- b
      fimse
   senao  // 2° caso: se A for menor que B e C
      se (a > b) ou (a > c) entao //(2° caso) e A for maior que um dos dois
         meio <- a //(2° caso) A fica no meio
         se (b > c) entao // (2° caso) Verificando o maior e o menor
            maior <- b
            menor <- c
         senao
            maior <- c
            menor <- b
         fimse
      senao //3° caso: mas se A for menor que todo mundo
         menor <- a //(3° caso)
         se (b > c) entao //(3° caso) Disputa de maior
            maior <- b
            meio <- c
         senao
            maior <- c
            meio <- b
         fimse

      fimse

   fimse

   escreval("Em ordem crescente...")
   escreva(menor, meio, maior)
fimalgoritmo
